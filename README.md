# DashboardStreamlit
Using python and Streamlit prepare a dashboard

# Create New venv in python
python -m venv myenv

# Activate venv --> myenv
myenv\Scripts\activate

# install all required lib. 
pip install -r requirements.txt

# use streamlit to run main.py
streamlit run main.py
